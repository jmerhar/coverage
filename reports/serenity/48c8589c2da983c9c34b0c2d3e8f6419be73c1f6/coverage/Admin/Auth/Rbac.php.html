7/**
8 * Role-based access control with a small, code-defined role → permissions map.
9 *
10 * This is a one/two-person operation, so the map is intentionally a constant —
11 * no DB-backed ACL and no permissions-management UI. It is designed to extend:
12 * later PRs add permission strings (e.g. 'bookings.confirm', 'reviews.moderate')
13 * and grant them to roles here. The wildcard '*' grants everything.
14 *
15 * Roles:
16 *  - owner:   full access, including user management ('users.manage').
17 *  - staff:   co-host — operational dashboard, logs, availability, bookings, channels, and review moderation.
18 *  - cleaner: the dashboard and the availability calendar (read-only).
19 *
20 * Every role can edit its own account ('profile.edit'); only the owner can
21 * manage other users ('users.manage', granted via the '*' wildcard).
22 */
23final class Rbac
24{
25    /** @var array<string, list<string>> */
26    private const array ROLES = [
27        'owner' => ['*'],
28        'staff' => [
29            'dashboard.view', 'logs.view',
30            'availability.view', 'availability.manage',
31            'reviews.view', 'reviews.moderate',
32            'bookings.view', 'bookings.manage',
33            'channels.view', 'channels.manage',
34            'profile.edit',
35        ],
36        'cleaner' => ['dashboard.view', 'availability.view', 'profile.edit'],
37    ];
38
39    /**
40     * Whether a role grants a permission.
41     *
42     * @param string|null $role The user's role (null when not logged in)
43     * @param string $permission Permission string, e.g. 'logs.view'
44     */
