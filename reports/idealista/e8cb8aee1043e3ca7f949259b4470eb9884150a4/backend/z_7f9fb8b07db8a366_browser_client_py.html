57def datadome_cookie_ttl_seconds(cookies: list[dict], now: float) -> float | None: 

58 """Remaining TTL (seconds) of the ``datadome`` cookie, or None if absent/session. 

59 

60 Diagnostic helper for the captcha cold-start investigation (see 

61 ``docs/scraper-captcha-cold-start.md``): the cookie expiring during a long 

62 idle gap is the suspected cause of pass-start captchas. 

63 

64 Args: 

65 cookies: Playwright cookie dicts (each with ``name`` and ``expires``). 

66 now: Current Unix time (seconds), for computing remaining TTL. 

67 

68 Returns: 

69 Seconds until the ``datadome`` cookie expires (negative if already 

70 expired), or None when there is no ``datadome`` cookie or it is a 

71 session cookie (browser-side ``expires`` of -1). Note: this is the 

72 browser-side expiry; DataDome may invalidate the cookie server-side 

73 sooner, so treat it as an upper bound. 

74 """ 

75 dd = next((c for c in cookies if c.get("name") == "datadome"), None) 

76 if dd is None: 

77 return None 

78 expires = dd.get("expires") 

79 if expires is None or expires < 0: 

80 return None 

81 return expires - now

179 async def get( 

180 self, url: str, *, delay: tuple[float, float] | None = None, archive: bool = True 

181 ) -> HttpResponse: 

182 """Navigate to URL, handle DataDome challenges, return structured response. 

183 

184 On first request, warms up by navigating to the homepage to obtain the 

185 DataDome cookie. Subsequent requests use the cached cookie and proceed 

186 directly to the target URL. 

187 

188 The HTML archive is write-only: pages are never read back here. Fetches 

189 always hit the live site. Successful responses are written to the archive 

190 (unless ``archive=False``) so the reparse job can re-extract fields later. 

191 

192 Args: 

193 url: The URL to navigate to. 

194 delay: Optional (min, max) seconds to sleep before network fetch. 

195 archive: Whether to write the fetched HTML to the archive. Set False 

196 for throwaway fetches (e.g. delisting 404/redirect checks). 

197 

198 Returns: 

199 HttpResponse with status_code, text, and final url. 

200 

201 Raises: 

202 CaptchaError: If DataDome challenge doesn't auto-resolve (full CAPTCHA). 

203 BlockedError: If the page is a block/rate-limit response. 

204 BrowserClientError: If navigation fails or browser is not initialized. 

205 """ 

206 if not self._page: 

207 raise BrowserClientError("Client not initialized — use async with") 

208 

209 # Refresh the liveness heartbeat on every fetch so a long active scrape 

210 # (search pagination + detail enrichment) keeps the worker healthy. 

211 beat() 

212 

213 # The archive is never read on normal runs — always fetch live. 

214 if delay is not None: 

215 await random_delay(*delay) 

216 

217 await self._maybe_rotate() 

218 await self._ensure_warmed_up() 

219 

220 response = await self._page.goto(url, wait_until=_WAIT_UNTIL, timeout=30000) 

221 html = await self._page.content() 

222 

223 # Check if DataDome presented a challenge 

224 if self._is_challenge(html): 

225 try: 

226 await self._wait_for_challenge_resolution("page") 

227 except CaptchaError: 

228 await self._escalate_unresolved_challenge() 

229 # After challenge resolution, re-navigate to get the actual page 

230 response = await self._page.goto( 

231 url, wait_until=_WAIT_UNTIL, timeout=30000 

232 ) 

233 html = await self._page.content() 

234 

235 status_code = response.status if response else 0 

236 

237 if is_blocked_page(html): 

238 raise BlockedError(f"Blocked fetching {url}") 

239 

240 if archive and 200 <= status_code < 400: 

241 self._dump_html(url, html) 

242 

243 final_url = self._page.url 

244 return HttpResponse(status_code=status_code, text=html, url=final_url)

344 async def _route_request(self, route: "Route") -> None: 

345 """Playwright route handler: abort render-weight sub-requests. 

346 

347 Drops images, fonts, media, and known-heavy third-party scripts (see 

348 :func:`should_block_request`) to reduce per-page CPU. DataDome's own 

349 traffic is always allowed so the anti-bot bypass is unaffected. 

350 

351 Fails open: any error deciding the verdict defaults to allowing the 

352 request, and every code path resolves the route (abort/continue). A 

353 route left unresolved would hang until the navigation timeout and stall 

354 ``networkidle``, so this never leaves one dangling. Errors from the 

355 resolve call itself (e.g. the request was already handled because the 

356 page navigated away) are swallowed — they are not actionable. 

357 """ 

358 try: 

359 block = should_block_request(route.request.resource_type, route.request.url) 

360 except Exception: 

361 logger.debug("Block decision failed; allowing request", exc_info=True) 

362 block = False 

363 try: 

364 if block: 

365 await route.abort() 

366 else: 

367 await route.continue_() 

368 except Exception: 

369 logger.debug("Route resolve failed (request likely superseded)", exc_info=True)

379 return f"{slug[:180]}_{url_hash}"

427

630 async def _wait_for_challenge_resolution(self, context_label: str = "page") -> None: 

631 """Wait for CloakBrowser to auto-resolve the DataDome challenge. 

632 

633 CloakBrowser executes the challenge JS natively (Canvas, WebGL, Audio 

634 fingerprinting) and the page reloads once the challenge is solved. Logs 

635 the resolution time (or timeout) tagged with ``context_label`` so cold 

636 warm-up challenges can be distinguished from in-pass ones (see 

637 ``docs/scraper-captcha-cold-start.md``). 

638 

639 Args: 

640 context_label: Where the challenge occurred — ``"warmup"`` (cold, 

641 pass-start) or ``"page"`` (mid-pass) — used only for logging. 

642 

643 Raises: 

644 CaptchaError: If the challenge doesn't resolve within timeout. 

645 """ 

646 logger.info("DataDome challenge (%s) detected — waiting for auto-resolution", context_label) 

647 elapsed = 0.0 

648 

649 while elapsed < _CHALLENGE_TIMEOUT: 

650 await asyncio.sleep(_CHALLENGE_POLL_INTERVAL) 

651 elapsed += _CHALLENGE_POLL_INTERVAL 

652 

653 html = await self._safe_content() 

654 if not self._is_challenge(html): 

655 logger.info("DataDome challenge (%s) resolved after %.1fs", context_label, elapsed) 

656 return 

657 

658 logger.warning( 

659 "DataDome challenge (%s) did not auto-resolve within %.1fs — escalating to CAPTCHA", 

660 context_label, elapsed, 

661 ) 

662 raise CaptchaError("DataDome challenge timeout — escalated to full CAPTCHA")
