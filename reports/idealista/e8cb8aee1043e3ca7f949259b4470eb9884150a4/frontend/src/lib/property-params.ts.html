/** Parse PropertySort from URLSearchParams. */
export function parseSortFromParams(params: URLSearchParams): PropertySort {
  const sortBy = params.get("sort_by");
  const sortDir = params.get("sort_dir");
  return {
    sort_by: (sortBy as PropertySort["sort_by"]) || "first_seen_at",
    sort_dir: (sortDir as PropertySort["sort_dir"]) || "desc",
  };
}
 
/** Parse PropertyPagination from URLSearchParams. */
export function parsePaginationFromParams(params: URLSearchParams): PropertyPagination {
  const limit = params.get("limit");
  const offset = params.get("offset");
  return {
    limit: limit ? Number(limit) : DEFAULT_LIMIT,
    offset: offset ? Number(offset) : 0,
  };
}
