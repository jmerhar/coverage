/** Derived per-run display metrics (not stored — computed from run counters). */
 
import type { ScrapeRun } from "../types/scrape-target";
 
/**
 * Detail pages fetched for listings that already existed (re-enrichments),
 * as opposed to first-time enrichment of brand-new listings.
 *
 * Not stored on the run — derived. Every detail fetch targets either a new
 * listing (counted in `listings_new`) or an existing one being re-enriched, so
 * `detail_pages_from_site = new + re-enriched`. Hence re-enriched ≈
 * `detail_pages_from_site − listings_new`. Exact when no fetches were skipped
 * (`detail_pages_skipped` is attributed to new listings here); clamped at 0 so
 * interrupted runs (where `listings_new` can exceed details actually fetched)
 * never show a negative. Returns 0 for runs that fetch no detail pages
 * (rentals, delist, reparse).
 */
export function reEnrichedCount(
  run: Pick<ScrapeRun, "detail_pages_from_site" | "listings_new">,
): number {
  return Math.max(0, run.detail_pages_from_site - run.listings_new);
}
