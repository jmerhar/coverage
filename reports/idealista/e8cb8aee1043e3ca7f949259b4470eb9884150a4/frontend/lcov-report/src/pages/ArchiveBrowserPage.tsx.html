/** Round up to a "nice" boundary for the slider max. */
function ceilToNice(hours: number): number {
  if (hours <= 1) return 1;
  if (hours <= 6) return 6;
  if (hours <= 12) return 12;
  if (hours <= 24) return 24;
  if (hours <= 48) return 48;
  if (hours <= 168) return 168;
  if (hours <= 720) return 720;
  return Math.ceil(hours / 720) * 720;
}
 
/** Generate meaningful mark positions for the slider. */
function generateMarks(max: number): Record<number, string> {
  const marks: Record<number, string> = { 0: "0h" };
  const candidates = [1, 6, 12, 24, 48, 168, 720];
  for (const h of candidates) {
    if (h < max) {
      marks[h] = formatMarkLabel(h);
    }
  }
  marks[max] = formatMarkLabel(max);
  return marks;
}
 
/** Human-readable label for a mark position. */
function formatMarkLabel(hours: number): string {
  if (hours === 0) return "0h";
  if (hours < 24) return `${hours}h`;
  if (hours === 168) return "7d";
  if (hours === 720) return "30d";
  if (hours % 720 === 0) return `${hours / 720}mo`;
  if (hours % 24 === 0) return `${hours / 24}d`;
  return `${hours}h`;
}
 
/** Format bytes as a human-readable string. */
function formatBytes(bytes: number): string {
  if (bytes < 1024) return `${bytes} B`;
  if (bytes < 1024 * 1024) return `${(bytes / 1024).toFixed(1)} KB`;
  return `${(bytes / (1024 * 1024)).toFixed(1)} MB`;
}
 
/** Format age in hours as a human-readable string. */
function formatAge(hours: number): string {
  if (hours < 1) return `${Math.round(hours * 60)}m ago`;
  if (hours < 24) return `${Math.round(hours)}h ago`;
  return `${Math.round(hours / 24)}d ago`;
}
 
/** Get a string ID for a group (used in expanded state and query keys). */
function groupKey(summary: ArchiveGroupSummary): string {
  return summary.target_id != null ? String(summary.target_id) : "ungrouped";
}
 
/** A single file row with link and optional checkbox. */
function FileRow({
  file,
  selected,
  onToggle,
  showCheckbox,
}: {
  file: ArchiveFileInfo;
  selected: boolean;
  onToggle: (filename: string) => void;
  showCheckbox: boolean;
}) {
  return (
    <div className="flex items-center gap-3 py-1 text-sm">
      {showCheckbox && (
        <input
          type="checkbox"
          checked={selected}
          onChange={() => onToggle(file.filename)}
          className="h-3.5 w-3.5 shrink-0 rounded border-gray-300 text-blue-600 dark:border-gray-600"
        />
      )}
      <a
        href={getArchiveFileUrl(file.filename)}
        target="_blank"
        rel="noopener noreferrer"
        className="min-w-0 flex-1 truncate text-blue-600 hover:underline dark:text-blue-400"
        title={file.filename}
      >
        {file.filename}
      </a>
      <span className="shrink-0 text-gray-500 dark:text-gray-400">{formatBytes(file.size_bytes)}</span>
      <span className="shrink-0 text-gray-400 dark:text-gray-500">{formatAge(file.age_hours)}</span>
    </div>
  );
}
 
/** File section (Search Pages / Detail Pages) with select-all checkbox. */
function FileSection({
  label,
  files,
  selected,
  onToggleFile,
  onToggleAll,
  showCheckboxes,
}: {
  label: string;
  files: ArchiveFileInfo[];
  selected: Set<string>;
  onToggleFile: (filename: string) => void;
  onToggleAll: (filenames: string[]) => void;
  showCheckboxes: boolean;
}) {
  if (files.length === 0) return null;
 
  const filenames = files.map((f) => f.filename);
  const allSelected = filenames.every((fn) => selected.has(fn));
  const someSelected = filenames.some((fn) => selected.has(fn));
 
  return (
    <div className="mt-3">
      <div className="flex items-center gap-2 border-b border-gray-100 pb-1 dark:border-gray-700">
        {showCheckboxes && (
          <input
            type="checkbox"
            checked={allSelected}
            ref={(el) => { if (el) el.indeterminate = someSelected && !allSelected; }}
            onChange={() => onToggleAll(filenames)}
            className="h-3.5 w-3.5 rounded border-gray-300 text-blue-600 dark:border-gray-600"
          />
        )}
        <span className="text-xs font-semibold uppercase text-gray-500 dark:text-gray-400">
          {label} ({files.length})
        </span>
      </div>
      <div className="ml-1 mt-1 space-y-0.5">
        {files.map((file) => (
          <FileRow
            key={file.filename}
            file={file}
            selected={selected.has(file.filename)}
            onToggle={onToggleFile}
            showCheckbox={showCheckboxes}
          />
        ))}
      </div>
    </div>
  );
}
 
/** Expandable group card that lazy-loads files on expand. */
function GroupCard({
  summary,
  expanded,
  onToggle,
  minAge,
  maxAge,
  selected,
  onToggleFile,
  onToggleAll,
  showCheckboxes,
}: {
  summary: ArchiveGroupSummary;
  expanded: boolean;
  onToggle: () => void;
  minAge: number | undefined;
  maxAge: number | undefined;
  selected: Set<string>;
  onToggleFile: (filename: string) => void;
  onToggleAll: (filenames: string[]) => void;
  showCheckboxes: boolean;
}) {
  const gid = groupKey(summary);
  const { data: group, isLoading } = useArchiveGroup(gid, minAge, maxAge, expanded);
  const totalFiles = summary.search_count + summary.detail_count;
 
  return (
    <div className="rounded-lg border bg-white shadow-sm dark:border-gray-700 dark:bg-gray-900">
      <button
        onClick={onToggle}
        className="flex w-full items-center gap-3 px-4 py-3 text-left"
      >
        <span className="text-gray-500 dark:text-gray-400">
          {expanded ? (
            <ChevronDown className="h-4 w-4" />
          ) : (
            <ChevronRight className="h-4 w-4" />
          )}

        <div className="flex-1">
          <span className="font-medium text-gray-900 dark:text-gray-100">{summary.target_label}</span>
          <span className="ml-2 text-sm text-gray-500 dark:text-gray-400">
            {totalFiles} files &middot; {formatBytes(summary.total_size_bytes)}
          </span>
        </div>
      </button>
      {expanded && (
        <div className="border-t px-4 pb-4 dark:border-gray-700">
          {isLoading ? (
            <div className="flex items-center gap-2 py-4 text-sm text-gray-500">
              <Loader2 className="h-4 w-4 animate-spin" />
              Loading files...
            </div>
          ) : group ? (
            <>
              <FileSection
                label="Search Pages"
                files={group.search_pages}
                selected={selected}
                onToggleFile={onToggleFile}
                onToggleAll={onToggleAll}
                showCheckboxes={showCheckboxes}
              />
              <FileSection
                label="Detail Pages"
                files={group.detail_pages}
                selected={selected}
                onToggleFile={onToggleFile}
                onToggleAll={onToggleAll}
                showCheckboxes={showCheckboxes}
              />
            </>
          ) : null}
        </div>
      )}
    </div>
  );
}
 
/** Collect all filenames from loaded groups in the query archive. */
function collectLoadedFilenames(
  queryClient: ReturnType<typeof useQueryClient>,
  expandedGroups: Set<string>,
  minAge: number | undefined,
  maxAge: number | undefined,
): string[] {
  const filenames: string[] = [];
  for (const gid of expandedGroups) {
    const data = queryClient.getQueryData<ArchiveGroup>(["archive-group", gid, minAge, maxAge]);
    if (data) {
      for (const f of data.search_pages) filenames.push(f.filename);
      for (const f of data.detail_pages) filenames.push(f.filename);
    }
  }
  return filenames;
}

        next.add(id);
      }

    });
  }, []);

      if (next.has(filename)) {
        next.delete(filename);
      } else {
        next.add(filename);
      }
      return next;
    });
  }, []);

        for (const fn of filenames) next.delete(fn);
      } else {
        for (const fn of filenames) next.add(fn);
      }
      return next;
    });
  }, []);

        onSuccess: () => setSelected((prev) => {
          const next = new Set(prev);
          for (const fn of filenames) next.delete(fn);
          return next;
        }),
      });
    }
  }

    if (filenames.length > 0) {
      deleteMutation.mutate(filenames, {
        onSuccess: () => setSelected(new Set()),
      });
    }
  }
 
  /** Describe current slider range as text. */
  function rangeLabel(): string {

    if (min === 0 && max >= sliderMax) return "All files";
    if (min === 0) return `Newer than ${formatMarkLabel(max)}`;
    if (max >= sliderMax) return `Older than ${formatMarkLabel(min)}`;
    return `${formatMarkLabel(min)} – ${formatMarkLabel(max)}`;
  }
 
  // Show loading state only on initial load (no previous data)
  if (isLoading && !summaries) {
    return (
      <div className="flex items-center justify-center py-12">
        <p className="text-gray-500 dark:text-gray-400">Loading archive files...</p>
      </div>
    );
  }
 
  if (error) {
    return (
      <div className="rounded-lg border border-red-200 bg-red-50 p-4 dark:border-red-800 dark:bg-red-900/20">
        <p className="text-red-700 dark:text-red-300">Failed to load archive files.</p>
      </div>
    );
  }
 
  const totalFiles = summaries?.reduce((sum, g) => sum + g.search_count + g.detail_count, 0) ?? 0;
  const totalSize = summaries?.reduce((sum, g) => sum + g.total_size_bytes, 0) ?? 0;
  const totalSearchFiles = summaries?.reduce((sum, g) => sum + g.search_count, 0) ?? 0;
  const totalDetailFiles = summaries?.reduce((sum, g) => sum + g.detail_count, 0) ?? 0;
 
  return (
    <div className="space-y-4">
      <div className="flex items-center justify-between">
        <div className="flex items-center gap-3">
          <div>
            <h2 className="text-2xl font-bold text-gray-900 dark:text-gray-100">HTML Archive</h2>
            <p className="mt-1 text-sm text-gray-600 dark:text-gray-400">
              {totalFiles} files &middot; {formatBytes(totalSize)} &middot; {summaries?.length ?? 0} groups
            </p>
          </div>
          {isFetching && (
            <Loader2 className="h-4 w-4 animate-spin text-gray-400" />
          )}
        </div>
        <div className="flex items-center gap-2">
          {isAdmin && totalSearchFiles > 0 && (
            <button
              onClick={() => handleBulkDeleteByType("search")}
              disabled={deleteByTypeMutation.isPending}
              className={`flex items-center gap-2 rounded-md px-3 py-2 text-sm font-medium text-white disabled:opacity-50 ${
                confirmingType === "search"
                  ? "bg-orange-600 hover:bg-orange-700"
                  : "bg-red-600 hover:bg-red-700"
              }`}
            >
              <Trash2 className="h-4 w-4" />
              {confirmingType === "search"
                ? `Confirm? (${totalSearchFiles} files)`
                : `Delete All Search Pages (${totalSearchFiles})`}
            </button>
          )}
          {isAdmin && totalDetailFiles > 0 && (
            <button
              onClick={() => handleBulkDeleteByType("detail")}
              disabled={deleteByTypeMutation.isPending}
              className={`flex items-center gap-2 rounded-md px-3 py-2 text-sm font-medium text-white disabled:opacity-50 ${
                confirmingType === "detail"
                  ? "bg-orange-600 hover:bg-orange-700"
                  : "bg-red-600 hover:bg-red-700"
              }`}
            >
              <Trash2 className="h-4 w-4" />
              {confirmingType === "detail"
                ? `Confirm? (${totalDetailFiles} files)`
                : `Delete All Detail Pages (${totalDetailFiles})`}
            </button>
          )}
          {isAdmin && selected.size > 0 && (
            <button
              onClick={handleDeleteSelected}
              disabled={deleteMutation.isPending}
              className="flex items-center gap-2 rounded-md bg-red-600 px-3 py-2 text-sm font-medium text-white hover:bg-red-700 disabled:opacity-50"
            >
              <Trash2 className="h-4 w-4" />
              Delete Selected ({selected.size})
            </button>
          )}
          {isAdmin && expandedGroups.size > 0 && (
            <button
              onClick={handleDeleteAllShown}
              disabled={deleteMutation.isPending}
              className="flex items-center gap-2 rounded-md bg-red-600 px-3 py-2 text-sm font-medium text-white hover:bg-red-700 disabled:opacity-50"
            >
              <Trash2 className="h-4 w-4" />
              Delete All Shown
            </button>
          )}
        </div>
      </div>
 
      <div className="rounded-lg border bg-white p-4 dark:border-gray-700 dark:bg-gray-900">
        <div className="space-y-3">
          <div className="flex items-center gap-4 text-sm">
            <label className="text-gray-600 dark:text-gray-400">Age filter:</label>
            <span className="font-medium text-gray-900 dark:text-gray-100">{rangeLabel()}</span>
          </div>
          <div className="px-2 pb-4">
            <Slider
              range
              min={0}
              max={sliderMax}
              value={sliderValue}
              marks={marks}
              onChange={(val) => setSliderValue(val as [number, number])}
              onChangeComplete={(val) => setCommittedValue(val as [number, number])}
              allowCross={false}
              ariaLabelForHandle={["Minimum age", "Maximum age"]}
            />
          </div>
        </div>
      </div>
 
      {totalFiles === 0 ? (
        <div className="rounded-lg border bg-white p-8 text-center dark:border-gray-700 dark:bg-gray-900">
          <p className="text-gray-500 dark:text-gray-400">No archived HTML files found.</p>
          <p className="mt-1 text-sm text-gray-400 dark:text-gray-500">
            {minAge != null || maxAge != null
              ? "Try adjusting the age filter."
              : "Enable CACHE_ARCHIVE_ENABLED and run the scraper to populate the archive."}
          </p>
        </div>
      ) : (
        summaries!.map((summary) => (
          <GroupCard
            key={groupKey(summary)}
            summary={summary}
            expanded={expandedGroups.has(groupKey(summary))}
            onToggle={() => toggleGroup(groupKey(summary))}
            minAge={minAge}
            maxAge={maxAge}
            selected={selected}
            onToggleFile={toggleFile}
            onToggleAll={toggleAll}
            showCheckboxes={isAdmin}
          />
        ))
      )}
    </div>
  );
}
