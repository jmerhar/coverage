/** Utilities for converting between PropertyFilters and saved filter criteria. */
 
import type { PropertyFilters, PropertySort } from "../types/property";
 
/** Convert PropertyFilters (and optional sort) to a criteria dict for storage (strips undefined/default values). */
export function filtersToCriteria(
  filters: PropertyFilters,
  sort?: PropertySort
): Record<string, unknown> {
  const criteria: Record<string, unknown> = {};
  for (const [key, value] of Object.entries(filters)) {
    if (value === undefined || value === null) continue;
    // Skip empty arrays
    if (Array.isArray(value) && value.length === 0) continue;
    criteria[key] = value;
  }
  // Include sort if non-default
  if (sort) {
    if (sort.sort_by !== "first_seen_at") criteria.sort_by = sort.sort_by;
    if (sort.sort_dir !== "desc") criteria.sort_dir = sort.sort_dir;
  }
  return criteria;
}
 
/** Convert stored criteria back to PropertyFilters. */
export function criteriaToFilters(
  criteria: Record<string, unknown>
): PropertyFilters {
  return { ...criteria } as PropertyFilters;
}
 
/** Convert stored criteria to URLSearchParams for navigation. */
export function criteriaToSearchParams(
  criteria: Record<string, unknown>
): URLSearchParams {
  const params = new URLSearchParams();
  for (const [key, value] of Object.entries(criteria)) {
    if (value === undefined || value === null) continue;
    if (Array.isArray(value)) {
      for (const item of value) {
        params.append(key, String(item));
      }
    } else if (typeof value === "boolean") {
      params.set(key, value ? "true" : "false");
    } else {
      params.set(key, String(value));
    }
  }
  return params;
}
 
/** Generate a human-readable summary of filter criteria. */
export function summarizeCriteria(criteria: Record<string, unknown>): string {
  const parts: string[] = [];
 
  if (criteria.price_min || criteria.price_max) {
    const min = criteria.price_min ? `€${Number(criteria.price_min).toLocaleString()}` : "";
    const max = criteria.price_max ? `€${Number(criteria.price_max).toLocaleString()}` : "";
    if (min && max) parts.push(`${min}–${max}`);
    else if (min) parts.push(`from ${min}`);
    else parts.push(`up to ${max}`);
  }
 
  if (criteria.bedrooms_min !== undefined) {
    const min = criteria.bedrooms_min as number;
    const max = criteria.bedrooms_max as number | undefined;
    if (max !== undefined && max === min) parts.push(`T${min}`);
    else if (max !== undefined) parts.push(`T${min}–T${max}`);
    else parts.push(`T${min}+`);
  }
 
  if (criteria.area_min || criteria.area_max) {
    const min = criteria.area_min ? `${criteria.area_min}` : "";
    const max = criteria.area_max ? `${criteria.area_max}` : "";
    if (min && max) parts.push(`${min}–${max} m²`);
    else if (min) parts.push(`from ${min} m²`);
    else parts.push(`up to ${max} m²`);
  }
 
  if (Array.isArray(criteria.property_types) && criteria.property_types.length > 0) {
    const types = (criteria.property_types as string[]).map(
      (t) => t.charAt(0).toUpperCase() + t.slice(1)
    );
    parts.push(types.join(", "));
  }
 
  if (criteria.gross_yield_min) {
    parts.push(`yield ≥${criteria.gross_yield_min}%`);
  }
 
  if (criteria.sort_by && criteria.sort_by !== "first_seen_at") {
    const label = String(criteria.sort_by).replace(/_/g, " ");
    const dir = criteria.sort_dir === "asc" ? "↑" : "↓";
    parts.push(`sort: ${label} ${dir}`);
  }
 
  return parts.length > 0 ? parts.join(" · ") : "All properties";
}
