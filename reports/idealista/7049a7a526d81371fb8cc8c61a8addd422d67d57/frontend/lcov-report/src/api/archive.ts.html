/** API functions for the archive browser. */
 
import { apiFetch } from "./client";
import type { ArchiveAgeRange, ArchiveGroup, ArchiveGroupSummary, DeleteArchiveResponse } from "../types/archive";
 
/** Fetch the maximum age of any file in the archive (for slider bounds). */
export function fetchArchiveAgeRange(): Promise<ArchiveAgeRange> {
  return apiFetch<ArchiveAgeRange>("/archive/age-range");
}
 
/** Fetch lightweight group summaries (no file arrays). */
export function fetchArchiveSummary(minAge?: number, maxAge?: number): Promise<ArchiveGroupSummary[]> {
  const params = new URLSearchParams();
  if (minAge != null) params.set("min_age_hours", String(minAge));
  if (maxAge != null) params.set("max_age_hours", String(maxAge));
  const qs = params.toString();
  return apiFetch<ArchiveGroupSummary[]>(`/archive/summary${qs ? `?${qs}` : ""}`);
}
 
/** Fetch full file list for a single archive group. */
export function fetchArchiveGroup(
  groupId: string,
  minAge?: number,
  maxAge?: number,
): Promise<ArchiveGroup> {
  const params = new URLSearchParams();
  if (minAge != null) params.set("min_age_hours", String(minAge));
  if (maxAge != null) params.set("max_age_hours", String(maxAge));
  const qs = params.toString();
  return apiFetch<ArchiveGroup>(`/archive/groups/${encodeURIComponent(groupId)}${qs ? `?${qs}` : ""}`);
}
 
/** Fetch all archive files grouped by scrape target (legacy endpoint). */
export function fetchArchiveFiles(): Promise<ArchiveGroup[]> {
  return apiFetch<ArchiveGroup[]>("/archive/files");
}
 
/** Delete specified archive files by filename. */
export function deleteArchiveFiles(filenames: string[]): Promise<DeleteArchiveResponse> {
  return apiFetch<DeleteArchiveResponse>("/archive/files", {
    method: "DELETE",
    body: JSON.stringify({ filenames }),
  });
}
 
/** Delete all archive files of a given type, optionally filtered by age. */
export function deleteArchiveFilesByType(
  fileType: "search" | "detail",
  minAge?: number,
  maxAge?: number,
): Promise<DeleteArchiveResponse> {
  const params = new URLSearchParams();
  params.set("file_type", fileType);
  if (minAge != null) params.set("min_age_hours", String(minAge));
  if (maxAge != null) params.set("max_age_hours", String(maxAge));
  return apiFetch<DeleteArchiveResponse>(`/archive/files-by-type?${params}`, { method: "DELETE" });
}
 
/** Get the URL to view a archived HTML file in the browser. */
export function getArchiveFileUrl(filename: string): string {
  return `/api/archive/files/${encodeURIComponent(filename)}`;
}
